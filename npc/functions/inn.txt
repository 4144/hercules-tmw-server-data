// INN

function	script	Inn	{
	mes "[" + @npcName$ + "]";
	mes "\"Would you like to rest? It's only " + @cost + " gp.\"";
	next;
	
	menu
		"Yes", -,
		"No", L_No;
	
	if (zeny < @cost) goto L_NoMoney;
	set zeny, zeny - @cost;
	heal 10000, 10000;

	mes "[" + @npcName$ + "]";
	mes "\"Sleep well!\"";
	next;
	close;
	
L_No:
	mes "[" + @npcName$ + "]";
	mes "\"See you.\"";
	close;
	
L_NoMoney:
	mes "[" + @npcName$ + "]";
	mes "\"You don't have enough money to stay here.\"";
	close;
}
